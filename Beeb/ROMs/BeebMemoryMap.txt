CLOCK	2.0	// CPU clock in MHz

//	Device			Name		Addr	Size	Wait		Misc		Comment
//												States		
//												[CPU Cycles]
ADD	DRAM			RAM32K		0000	8000	0						// 32kB RAM	0000 -	7fff
ADD	ROM 			BASIC		8000	4000	0			BASIC2.rom	// Paged ROM #15 (IC101 SE)	8000 - bfff
ADD	ROM				MMFS		8000	4000	0			MMFS.rom	// Paged ROM #13 (IC52 NW)	8000 - bfff
ADD	ROM				OS			c000	4000	0			OS12.rom	// OS (IC51)		 		c000 -	ffff
GAP					OS			fc00	0300							// Gap in OS ROM space between fc00-ff00	
//
// SHEILA	fe00 -	feff
//
ADD	CRTC6845		CRTC		fe00	0008	1						// H6845SP CRTC
ADD	TT5050			TCG				1									// Teletext Character Generator			
ADD	ACIA6850		ACIA		fe08	0008	1			1.0			// ACIA 6850
ADD	BeebSerULA		SerialULA	fe10	0010	1						// Beeb Serial ULA
ADD	BeebVideoULA	VDU			fe20	0010	1						// Beeb Video ULA
ADD	BeebRomSel		ROMSel		fe30	0010	0
ADD	VIA6522			SYS_VIA		fe40	0020	1			1.0			// IC3	fe40 -	fe4f; 1 Mhz Clock
ADD	VIA6522			USER_VIA	fe60	0020	1			1.0			// IC69	fe60 -	fe6f; 1 MHz Clock
//ADD	ADC7002		ADC			fec0	0003	1						// ADC Converter

ADD	BEEBKB			KB		00											// startup options (DIP switches)
ADD	CPU_6502		CPU													// Microprocessor 6502
ADD	TAPREC			TAPREC												// Standard Tape Recorder
ADD	TI4689			SOUND
ADD	BEEBVIALATCH	VIAL												// System VIA Latch with 8 outputs, each updated by writing to Port B bits 3:0
ADD SD_CARD			MMFS_CARD							Disc.mmb		// SD Card formated with Disc.mmb

//
//	Device connections
//
//	Specifies (when applicable) each device's connection to other
//	devices. Only one port at a time can be specified per line.
//	For a multiple-bit port, the bits (starting with the high bit no)
//	to be connected shall be specified unless all bits shall be
//	connected.

// Sound chip connections
CONNECT:P 	SYS_VIA:PB;3;0	VIAL:CTRL	// PB3:0 => CTRL
CONNECT:P	VIAL:Q;0		SOUND:WE	// Q0 => WE
CONNECT:P	SYS_VIA:PA		SOUND:D		// PA7:0 => D

// Video Display Unit connections
CONNECT:I	CRTC:RA;0			TCG:CRS
CONNECT:P	CRTC:VS				TCG:DEW
CONNECT:IP	CRTC:HS				TCG:GLR
CONNECT:P	CRTC:DISPTMG		TCG:LOSE
CONNECT		CRTC:DISPTMG		VDU:DISEN
CONNECT		CRTC:CUDISP			VDU:CURSOR
CONNECT		CRTC:RA;3;0			VDU:RA
CONNECT		CRTC:VS				SYS_VIA:CA;0	// VS => CA1
CONNECT:P	VIAL:Q;5;4			VDU:C
CONNECT 	VDU:CRTC_CLK		CRTC:CLK

// Paged ROMs
CONNECT 	ROMSel:SE			BASIC:CS
CONNECT		ROMSel:NW			MMFS:CS

//
// SD Card for MMFS connected to the User VIA
//
// VIA CB1/CB2 are used for clock/data during shift in to VIA amd PB1/PB0 for clock/data during shift out from VIA
CONNECT:P	USER_VIA:CB;0			MMFS_CARD:CLK		// Clock - CB1 used to provide shift clock to the card (shift in to VIA as well as shift out from VIA)
CONNECT:P	USER_VIA:PB;1			USER_VIA:CB;0		// Clock - PB1 used to generate the CB1 shift clock (shift out from VIA only; otherwise PB1 is input)
CONNECT:P	USER_VIA:PB;0			MMFS_CARD:MOSI		// Serial Data In sent from PB0 (VIA PB1 output connected to CB1 shift clock input)
CONNECT:P	MMFS_CARD:MISO			USER_VIA:CB;1		// Serial Data Out received on CB2 (VIA timer T2 generates CB1 shift clock output)
INIT		MMFS_CARD:SEL			0					// Select hard-wired to be active


// VIA IRQ
CONNECT 	SYS_VIA:IRQ			CPU:IRQ			// VIA generates 100 Hz timer interrupts and more
CONNECT		USER_VIA:IRQ		CPU:IRQ			// User-define usage (i.e.not used by the BBC MOS)

// Keyboard connections
CONNECT:P	VIAL:Q;3			KB:ENA			// Connect System VIA Latch Q3 (value set by writing to System VIA Port B) to KB ENA
CONNECT:P 	SYS_VIA:PA;3;0		KB:COL_SEL		// PA3:0 => COL_SEL
CONNECT:P 	SYS_VIA:PA;6;4		KB:ROW_SEL		// PA6:4 => ROW_SEL
CONNECT 	KB:ROW				SYS_VIA:PA;7	// ROW => PA7
CONNECT 	KB:PRESSED			SYS_VIA:CA;1	// PRESSED = CA2
CONNECT 	KB:BREAK			CPU:RESET		// Connect BREAK key to uC RESET input
CONNECT 	KB:BREAK			SYS_VIA:RESET
CONNECT 	KB:BREAK			USER_VIA:RESET


// CUTS <=> Tape Recorder Connections
CONNECT 	TAPREC:CAS_IN		SerialULA:CAS_IN	// Connect Serial ULA (CUTS) and tape recorder - CAS IN
CONNECT 	SerialULA:CAS_OUT	TAPREC:CAS_OUT		// Connect Serial ULA (CUTS) and tape recorder - CAS OUT
CONNECT:P	ACIA:TxD			SerialULA:TxD		// Connect the ACIA Tx data to the Serial ULA with direct processing on update
CONNECT:P	ACIA:RTS			SerialULA:RTSI		// Connect the ACIA RTS to the Serial ULA with direct processing on update
CONNECT		SerialULA:RxD		ACIA:RxD			// Connect the Serial ULA Rx data to the ACIA
CONNECT:P	SerialULA:CTSO		ACIA:CTS			// Connect the Serial ULA CTS to the ACIA with direct processing on update
CONNECT:P	SerialULA:DCD		ACIA:DCD			// Connect the Serial ULA DCD to the ACIA with direct processing on update
CONNECT		ACIA:IRQ			CPU:IRQ

//
//	Device scheduling
//
SCHED	KB			HLINE
SCHED	CRTC		NONE
TRIGGER	CRTC		CALL	VDU							// Trigger the CRTC explictly by VDU function calls
SCHED	TCG			NONE
TRIGGER	TCG			CALL	VDU							// Trigger the TCG explictly by VDU function calls
TRIGGER ACIA		CALL	SerialULA					// Trigger the ACIA explictly by Serial ULA function calls
SCHED	ACIA		INSTR
SCHED	SerialULA	INSTR
SCHED	VDU			NONE
SCHED	VIAL		NONE
SCHED	SOUND		HLINE
SCHED	MMFS_CARD	NONE

//
INIT	KB:SW		0										// Keyboard switches; set to start with Mode 7 (value 0)
INIT	VDU:INV		1										// Inverted screen for non-teletext modes OFF (value 1) or ON (value 0)
