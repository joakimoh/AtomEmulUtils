//
//	Device instances
//
//	Devices are either a microprocessor, a memory-mapped peripheral
//	or a non-memory mapped peripheral.
//
//	Only a memory-mapped device has address and size properties.
//
//	Only a ROM has a file reference (the ROM data).
//
//	Supported devices are:
//		RAM:		RAM of any size (within the 64k address space)
//		ROM:		ROM of any size (within the 64k address space)
//		VDU6847:	A 6847 Video Data Unit (memory-mapped)
//		PIA8255		An 8255 PIA (memory-mapped)
//		UC6052		A 6502 microprocessor (obviously NOT memory-mapped)
//		ATOMKB		An Acorn Atom Keyboard (NOT memory-mapped)
//		ATOMCAS		An Acorn Atom Cassette Interface (NOT memory-mapped)
//		TAPREC		A Tape Recorder emulation (CSW files read/written)
//
//	Device	Unique name	Addr	Size
//
	RAM	BLOCK_ZERO	0000	0400				// Block Zero RAM
	VDU6847	VDU		0800	0100 8000			// VDU CRT Controller 6847
	RAM	PROGRAM		2800	5800				// Program RAM - 22K
	RAM	GRAPHICS	8000	1800				// Graphics RAM - 6K
	ROM	WILLOW		A000	1000 Willow.rom			// Willow Utility ROM
	PIA8255	PIA		b000	0004 				// PIA 8255
	VIA6522	VIA		b800	0010 				// VIA 6522
	ROM	BASIC		c000	1000 Atom_basic.rom		// BASIC ROM
	ROM	FP		d000	1000 Atom_FloatingPoint.rom	// Floating-point ROM
	ROM	KERNEL		f000	1000 Atom_Kernel.rom		// CAS ROM
	ATOMKB	ATOMKB							// Acorn Atom keyboard
	ATOMCAS	CUTS							// CUTS Cassette Interface
	UC6502	UC							// Microcontroller 6502
	TAPREC	TAPREC							// Standard Tape Recorder
	ATOMSP	SPEAKER							// Acorn Atom Speaker
//
//	Device connections
//
//	Specifies (when applicable) each device's connection to other
//	devices. Only one port at a time can be specified per line.
//	For a multiple-bit port, the bits (starting with the high bit no)
//	to be connected shall be specified unless all bits shall be
//	connected.
//

// PIA <=> VDU Connections							
CONNECT	VDU:FS		PIA:PortC;7			
CONNECT	PIA:PortA;4	VDU:A/G
CONNECT	VDU:Din;7	VDU:INV
CONNECT	VDU:Din;6	VDU:A/S
CONNECT	VDU:Din;6	VDU:INT/EXT
CONNECT	PIA:PortA;7;5	VDU:GM

// ATOMKB <=> PIA Connections
CONNECT ATOMKB:COL_L	PIA:PortB	// non-modifiers keys + SHIFT + CTRL
CONNECT ATOMKB:COL_H;0	PIA:PortC;6	// REPEAT key
CONNECT PIA:PortA;3;0	ATOMKB:ROW

// ATOMKB BREAK key <=> device RESET Connections
CONNECT ATOMKB:COL_H;1	UC:RESET	// BREAK key => RESET of microprocessor
CONNECT ATOMKB:COL_H;1	PIA:RESET	// BREAK key => RESET of PIA 8255
CONNECT ATOMKB:COL_H;1	VIA:RESET	// BREAK key => RESET of VIA 6522
CONNECT ATOMKB:COL_H;1	VDU:RESET	// BREAK key => RESET of VDU 6847

// VIA <=> Microcontroller Connections
CONNECT VIA:IRQ		UC:IRQ

// PIA <=> CUTS Connections
CONNECT PIA:PortC;0	CUTS:TAPE_OUT	// tape out
CONNECT PIA:PortC;1	CUTS:ENA_TONE	// enable tone to cassette output
CONNECT CUTS:TONE	PIA:PortC;4	// 2.4 Khz tone
CONNECT CUTS:TAPE_IN	PIA:PortC;5	// cassette input

// CUTS <=> Tape Recorder Connections
CONNECT TAPREC:CAS_IN	CUTS:CAS_IN	// connect CUTS and tape recorder - CAS IN
CONNECT CUTS:CAS_OUT	TAPREC:CAS_OUT	// connect CUTS and tape recorder - CAS OUT

// PIA <=> Speaker Connections
CONNECT	PIA:PortC;2	SPEAKER:OUT
